```java
class ForTest{
    public static void main(String[] args){
        int a,b = 1;
        while (b < 10){
            for (a = 1;a <= b;a++){
                System.out.print(a + "x" + b + "=" + (a * b) + " ");
            }
            System.out.print("\n");
            b++;
        }
    }
}
```
